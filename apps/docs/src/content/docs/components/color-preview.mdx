## Usage

<Tabs syncKey="framework">
  <TabItem label="Vue">
  ```ts
  import { ColorPreview } from '@hueycolor/vue'
  ```
  ```vue
  <script setup>
  import { ref } from 'vue'
  import { HueyRoot, ColorPreview, hueyColor } from '@hueycolor/vue'

  const color = ref(hueyColor('#1A326380'))
  </script>

  <template>
    <HueyRoot v-model="color">
      <ColorPreview />
    </HueyRoot>
  </template>
  ```
  </TabItem>
  <TabItem label="Svelte">
  ```ts
  import { ColorPreview } from '@hueycolor/svelte'
  ```
  ```svelte
  <script>
  import { HueyRoot, ColorPreview, hueyColor } from '@hueycolor/svelte'

  let color = $state(hueyColor('#1A326380'))
  </script>

  <HueyRoot bind:color>
    <ColorPreview />
  </HueyRoot>
  ```
  </TabItem>
</Tabs>
